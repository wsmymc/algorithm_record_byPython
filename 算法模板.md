```C++
{
    if (l>=r) return;
    int x=q[(l+r)/2], i= l-1,j =r+1;
    while(i<j)
    {
        do i++;while (q[i]<x);
        do j--;
        while (q[j]>x);
        if(i<j) swap(q[i],q[j]);
    }
    quick_sort(q,l,j);
    quick_sort(q,j+1,r);
```

### 前缀和和差分

#### 一维前缀和

```C++
S[i] = a[1] + a[2] + ... a[i]
a[l] + ... + a[r] = S[r] - S[l - 1]
```

#### 二维前缀和

```C++
S[i, j] = 第i行j列格子左上部分所有元素的和
以(x1, y1)为左上角，(x2, y2)为右下角的子矩阵的和为：
S[x2, y2] - S[x1 - 1, y2] - S[x2, y1 - 1] + S[x1 - 1, y1 - 1]
```

### 区间合并

```C++
// 将所有存在交集的区间合并
void merge(vector<PII> &segs)
{
    vector<PII> res;

    sort(segs.begin(), segs.end());//先排序

    int st = -2e9, ed = -2e9;//预设一个起点、终点
    for (auto seg : segs)
        if (ed < seg.first) // 如果端点小于下一节的起点，那么重启一个区间
        {
            // 首先不能是之前的那个“虚区间”
            if (st != -2e9) res.push_back({st, ed});
            //更新起点，终点
            st = seg.first, ed = seg.second;
        }
        else ed = max(ed, seg.second); // 否则的话，区间合并，右边界取两者之间最大值

    if (st != -2e9) res.push_back({st, ed});

    segs = res;
}
```

* 插入一个数: heap[++size] =x ; up(size);
* 求一个集合中的最小值:heap[1]
* 删除最小值: heap[1]=heap[size]; size--;down(1);

* 删除任意一个元素：heap[k] = heap[size]; down(k);up(k);

```C++

// ph[k]存储第k个插入的点在堆中的位置
// hp[k]存储堆中下标是k的点是第几个插入的
int h[N], ph[N], hp[N], size;

// 交换两个点，及其映射关系
void heap_swap(int a, int b)
{
    swap(ph[hp[a]],ph[hp[b]]);
    swap(hp[a], hp[b]);
    swap(h[a], h[b]); 
}

void down(int u)
{
    int t = u;
    if (u * 2 <= size && h[u * 2] < h[t]) t = u * 2;
    if (u * 2 + 1 <= size && h[u * 2 + 1] < h[t]) t = u * 2 + 1;
    if (u != t)
    {
        heap_swap(u, t);
        down(t);
    }
}
```

#### 字符串哈希 

* 用于快速判断两个字符串是否相等
* 感觉有点像是纯数学结论，主要的get公式怎么得到的？

```C++
核心思想：将字符串看成P进制数，P的经验值是131或13331，取这两个值的冲突概率低
小技巧：取模的数用2^64，这样直接用unsigned long long存储，溢出的结果就是取模的结果

typedef unsigned long long ULL;
ULL h[N], p[N]; // h[k]存储字符串前k个字母的哈希值, p[k]存储 P^k mod 2^64

// 初始化
p[0] = 1;
for (int i = 1; i <= n; i ++ )
{
    h[i] = h[i - 1] * P + str[i];
    p[i] = p[i - 1] * P;
}

// 计算子串 str[l ~ r] 的哈希值
ULL get(int l, int r)
{
    return h[r] - h[l - 1] * p[r - l + 1];
}
```
